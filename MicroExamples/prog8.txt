LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
COUNTER EQU h'020'
BANKSEL TRISB ; Go to Bank1
CLRF TRISB ; Clear Trisb
BANKSEL PORTB ; Go to Bank0
CLRF PORTB ; Clear Portb
STARTLABEL 
CLRF PORTB ; Clear Portb
CLRF COUNTER ; Clear Counter
LOOPLABEL
MOVF COUNTER, W ; Load counter to W
MOVWF PORTB ; Assign W to Portb
INCF COUNTER, F ; Counter=Counter+1
MOVLW h'F5' ; Load h'F5' to W
SUBWF COUNTER,W ; W = Counter-W
BTFSS STATUS,2 ; STATUS bit2 =1?
GOTO LOOPLABEL ;No
FINALLABEL     ;Yes
GOTO FINALLABEL
END
