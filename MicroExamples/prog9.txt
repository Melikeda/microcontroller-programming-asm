LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
COUNTER1 EQU H'20'
COUNTER2 EQU H'21'
CLRF PORTB
BANKSEL TRISB
CLRF TRISB
BANKSEL PORTB


REPEATLABEL
MOVLW h'00'
MOVWF PORTB
CALL DELAYLABEL
MOVLW h'FF'
MOVWF PORTB
CALL DELAYLABEL
GOTO REPEATLABEL


DELAYLABEL
MOVLW h'FF'
MOVWF COUNTER1


LOOP1
MOVLW h'FF'
MOVWF COUNTER2

LOOP2
DECFSZ COUNTER2, F
GOTO LOOP2
DECFSZ COUNTER1,F
GOTO LOOP1
RETURN
END
