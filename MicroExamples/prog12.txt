LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
COUNTER1 EQU H'20'
COUNTER2 EQU H'21'
CLRF PORTB
BCF STATUS,0 ;Reset Carry Flag
BANKSEL TRISB
CLRF TRISB
BANKSEL PORTB

STARTLABEL
MOVLW h'01'
MOVWF PORTB

LEFTLABEL
CALL DELAYSUBPROGRAMLABEL
RLF PORTB, F
BTFSS PORTB,7
GOTO LEFTLABEL

RIGHTLABEL
CALL DELAYSUBPROGRAMLABEL
RRF PORTB,F
BTFSS PORTB,0
GOTO RIGHTLABEL
GOTO LEFTLABEL

DELAYSUBPROGRAMLABEL
MOVLW h'FF'
MOVWF COUNTER1

LOOP1
MOVLW h'FF'
MOVWF COUNTER2

LOOP2
DECFSZ COUNTER2,F
GOTO LOOP2
DECFSZ COUNTER1, F
GOTO LOOP1
RETURN
END
