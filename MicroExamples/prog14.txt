LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
ORG h'00'
CLRF PORTB
CLRF PORTA
BANKSEL TRISB
CLRF TRISB
MOVLW b'11110111'
MOVWF TRISA
BANKSEL PORTB

STARTLABEL
MOVLW h'5A'
ADDLW h'53'
MOVWF PORTB
BTFSC STATUS,0
BSF PORTA,3

LOOPLABEL
GOTO LOOPLABEL
END
