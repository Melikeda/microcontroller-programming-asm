LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
CBLOCK 20h
AL,AH,BL,BH
ENDC

ORG h'00'
CLRF PORTB
BANKSEL TRISB
CLRF TRISB
MOVLW H'FF'
MOVWF TRISA
BANKSEL PORTB
MOVLW h'07'
MOVWF CMCON

STARTLABEL
MOVLW h'A3'
MOVWF AL
MOVLW h'61'
MOVWF AH
MOVLW h'E0'
MOVWF BL
MOVLW h'2E'
MOVWF BH

ADDITIONLABEL
MOVF AL,W 
ADDWF BL,F  ;BL=BL+W
BTFSC STATUS,0 ; Carry==1?
INCF BH,F
MOVF AH,W
ADDWF BH,F ; BH=BH+W

SHOWLOWERBYTESLABEL
MOVF BL,W
MOVWF PORTB

TEST_RA0
BTFSC PORTA,0
GOTO TEST_RA0

SHOWUPPERBYTESLABEL
MOVF BH,W
MOVWF PORTB

LOOPLABEL
GOTO LOOPLABEL
END
