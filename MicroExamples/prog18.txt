LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
CBLOCK 20h
AL,AH,BL,BH
ENDC
ORG h'00'
CLRF PORTB
BANKSEL TRISB
CLRF TRISB
MOVLW H'FF'
MOVWF TRISA
BANKSEL PORTB
MOVLW h'07'
MOVWF CMCON

STARTLABEL
MOVLW h'04'
MOVWF AL
MOVLW h'00'
MOVWF AH
MOVLW h'01'
MOVWF BL
MOVLW h'00'
MOVWF BH

SUBTRACTIONLABEL
MOVF BL,W
SUBWF AL,F
BTFSS STATUS,0
DECF AH,F
MOVF BH,W
SUBWF AH,F

SHOWLOWERBYTES
MOVF AL,W
MOVWF PORTB

TEST_RA0
BTFSC PORTA,0
GOTO TEST_RA0
MOVF AH,W
MOVWF PORTB

LOOPLABEL
GOTO LOOPLABEL
END
