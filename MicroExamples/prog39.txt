;===PROG39.ASM
	LIST	P=16F628A
	INCLUDE	"P16F628A.INC"
	__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF & _LVP_OFF & _CP_OFF
;-----------------------------------------------------------
	COUNTER EQU H'20'
	ADR EQU H'20'
	DAT EQU H'20'
	
	BANKSEL TRISB
	CLRF TRISB
	MOVLW h'FF'
	MOVWF TRISA
	BANKSEL PORTB
	MOVLW h'07'
	MOVWF CMCON
	CLRF PORTB
	
	RA0_TEST
	BTFSC PORTA,0
	GOTO RA1_TEST
	CALL DELAYLABEL
	MOVLW h'05'
	MOVWF ADR
	MOVLW h'8F'
	MOVWF DAT
	CALL WRITE
	
	RA1_TEST
	BTFSC PORTA,1
	GOTO RA2_TEST
	CALL DELAYLABEL
	MOVLW h'05'
	MOVWF ADR
	MOVLW h'3D'
	MOVWF DAT
	CALL WRITE
	
	RA2_TEST
	BTFSC PORTA,2
	GOTO RA0_TEST
	CALL READ
	GOTO RA0_TEST
	
	READ
	MOVLW h'05'
	MOVWF ADR
	BANKSEL EEADR
	MOVWF EEADR
	BSF EECON1,RD
	MOVF EEDATA,W
	BANKSEL PORTB
	MOVWF PORTB
	RETURN
	
	WRITE
	BANKSEL EEADR
	MOVF ADR,W
	MOVWF EEADR
	MOVF DAT,W
	MOVWF EEDATA
	
	BANKSEL PIR1
	BCF PIR1,EEIF
	BANKSEL EECON1
	BSF EECON1,WREN
	BCF INTCON,GIE
	
	MOVLW h'55'
	MOVWF EECON2
	MOVLW h'AA'
	MOVWF EECON2
	BSF EECON1,WR; start the writing operation
	BANKSEL PIR1
	WAIT
	BTFSS PIR1,EEIF ;checks the writing operation
	GOTO WAIT
	
	BCF PIR1,EEIF
	BANKSEL EECON1
	BCF EECON1,WREN ;prevent the writing operation
	BSF INTCON,GIE
	BANKSEL PORTB
	RETURN
	
	DELAYLABEL
	MOVLW h'55'
	MOVWF COUNTER
	WAIT2
	DECFSZ COUNTER
	GOTO WAIT2
	RETURN
	END
