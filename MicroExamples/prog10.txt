LIST P=16F628A
INCLUDE "P16F628A.INC"
__CONFIG _INTRC_OSC_NOCLKOUT & _WDT_OFF & _PWRTE_ON & _MCLRE_ON & _BODEN_OFF& _LVP_OFF & _CP_OFF
COUNTER1 EQU H'20'
COUNTER2 EQU H'21'
ORG h'00'
CLRF PORTB
BANKSEL TRISB
CLRF TRISB
MOVLW H'FF'
MOVWF TRISA
BANKSEL PORTB
MOVLW h'07'
MOVWF CMCON

RA0_TEST
BTFSC PORTA,0
GOTO RA2_TEST ; No

INCREASELABEL
MOVLW h'FF'    
SUBWF PORTB,W  ; W=PORTB-W
BTFSS STATUS,2 ;W=PORTB=255?
INCF PORTB,F   ;No
CALL DELAYLABEL ;Yes
GOTO RA0_TEST



RA2_TEST
BTFSC PORTA,2
GOTO RA0_TEST


DECREASELABEL
MOVF PORTB,W
BTFSS STATUS,2
DECF PORTB,F
CALL DELAYLABEL
GOTO RA0_TEST

DELAYLABEL
MOVLW h'FF'    ;Outer Loop (M)
MOVWF COUNTER1

LOOP1
MOVLW h'FF'    ;Inner Loop (M)
MOVWF COUNTER2

LOOP2
DECFSZ COUNTER2, F
GOTO LOOP2
DECFSZ COUNTER1,F
GOTO LOOP1
RETURN
END
